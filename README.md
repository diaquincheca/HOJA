\cvsubsection{Hospital Universitario de la Samaritana\\
\textnormal{Servicio Farmacéutico – Gestión asistencial y calidad}}

\textbf{Acciones:}
\begin{itemize}
    \item Ejecución de controles de almacenamiento, validación de cadena de frío e inspecciones internas.
    \item Actualización de POE relacionados con dispensación, custodia, narcóticos y medicamentos de alto riesgo.
    \item Aplicación de análisis de riesgos y gestión de incidentes para seguridad del paciente.
\end{itemize}

\textbf{Resultados:}
\begin{itemize}
    \item Reducción del 30\% en desviaciones menores de almacenamiento.

\end{itemize}
